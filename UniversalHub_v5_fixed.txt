local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local UIS = game:GetService("UserInputService")
local lp = Players.LocalPlayer
local cam = game.Workspace.CurrentCamera
local efectos = {}
local conexiones = {}

local shaderVals = {
    bloom = 1.0,
    saturacion = 0.0,
    contraste = 0.0,
    brillo = 0.0,
    rayossol = 0.0,
    blur = 0,
}

local function limpiarConexion(tag)
    if conexiones[tag] then conexiones[tag]:Disconnect(); conexiones[tag] = nil end
end

local function nuevoEfecto(clase)
    local e = Instance.new(clase, Lighting)
    table.insert(efectos, e)
    return e
end

local function limpiarShaders()
    for _, e in ipairs(efectos) do if e and e.Parent then e:Destroy() end end
    efectos = {}
end

local function aplicarSliders()
    for _, e in ipairs(efectos) do
        if e:IsA("BloomEffect") then
            e.Intensity = shaderVals.bloom
        elseif e:IsA("ColorCorrectionEffect") then
            e.Saturation = shaderVals.saturacion
            e.Contrast = shaderVals.contraste
            e.Brightness = shaderVals.brillo
        elseif e:IsA("SunRaysEffect") then
            e.Intensity = shaderVals.rayossol
        elseif e:IsA("BlurEffect") then
            e.Size = shaderVals.blur
        end
    end
end

PlayerTab:CreateToggle({
    Name = "Fly (WASD + Space/Shift)",
    CurrentValue = false,
    Flag = "Fly",
    Callback = function(val)
        limpiarConexion("fly")
        limpiarConexion("fly_respawn")

        local function detenerFly()
            limpiarConexion("fly")
            local char = lp.Character
            if not char then return end
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChild("Humanoid")
            if hrp then
                -- Primero poner velocidad en 0 antes de destruir
                local bv = hrp:FindFirstChild("FlyVelocity")
                if bv then bv.Velocity = Vector3.zero; bv:Destroy() end
                local bg = hrp:FindFirstChild("FlyGyro")
                if bg then bg:Destroy() end
            end
            if hum then
                hum.PlatformStand = false
            end
        end

        local function iniciarFly()
            detenerFly()
            local char = lp.Character
            if not char then return end
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChild("Humanoid")
            if not hrp or not hum then return end

            hum.PlatformStand = true

            local bg = Instance.new("BodyGyro")
            bg.Name = "FlyGyro"
            bg.MaxTorque = Vector3.new(1e9, 1e9, 1e9)
            bg.D = 50; bg.P = 1e4
            bg.CFrame = hrp.CFrame
            bg.Parent = hrp

            local bv = Instance.new("BodyVelocity")
            bv.Name = "FlyVelocity"
            bv.MaxForce = Vector3.new(1e9, 1e9, 1e9)
            bv.Velocity = Vector3.zero
            bv.Parent = hrp

            conexiones["fly"] = RunService.Heartbeat:Connect(function()
                if not val or not hrp or not hrp.Parent then return end
                local speed = 60
                local dir = Vector3.zero
                if UIS:IsKeyDown(Enum.KeyCode.W) then dir += cam.CFrame.LookVector end
                if UIS:IsKeyDown(Enum.KeyCode.S) then dir -= cam.CFrame.LookVector end
                if UIS:IsKeyDown(Enum.KeyCode.A) then dir -= cam.CFrame.RightVector end
                if UIS:IsKeyDown(Enum.KeyCode.D) then dir += cam.CFrame.RightVector end
                if UIS:IsKeyDown(Enum.KeyCode.Space) then dir += Vector3.new(0, 1, 0) end
                if UIS:IsKeyDown(Enum.KeyCode.LeftShift) then dir -= Vector3.new(0, 1, 0) end
                bv.Velocity = dir.Magnitude > 0 and dir.Unit * speed or Vector3.zero
                bg.CFrame = cam.CFrame
            end)
        end

        if val then
            iniciarFly()
            conexiones["fly_respawn"] = lp.CharacterAdded:Connect(function()
                task.wait(0.5)
                if val then iniciarFly() end
            end)
        else
            detenerFly()
        end
    end,
})

local PlayersTab = Window:CreateTab("🎮 Jugadores", 4483362458)

local targetPlayer = nil

local function getPlayerNames()
    local names = {}
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= lp then table.insert(names, p.Name) end
    end
    return names
end

PlayersTab:CreateDropdown({
    Name = "Seleccionar jugador",
    Options = getPlayerNames(),
    CurrentOption = {},
    Flag = "TargetPlayer",
    MultipleOptions = false,
    Callback = function(val)
        targetPlayer = Players:FindFirstChild(type(val) == "table" and val[1] or val)
    end,
})

PlayersTab:CreateButton({
    Name = "Ver info del jugador",
    Callback = function()
        if not targetPlayer then
            Rayfield:Notify({ Title = "Error", Content = "Selecciona un jugador primero", Duration = 3 })
            return
        end
        local char = targetPlayer.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChild("Humanoid")
        local ping = math.floor(targetPlayer:GetNetworkPing() * 1000)
        local dist = (hrp and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart"))
            and math.floor((hrp.Position - lp.Character.HumanoidRootPart.Position).Magnitude).."m" or "N/A"
        local hp = hum and math.floor(hum.Health) or "N/A"
        local maxHp = hum and math.floor(hum.MaxHealth) or "N/A"
        Rayfield:Notify({
            Title = "👤 " .. targetPlayer.Name,
            Content = "❤ HP: "..hp.."/"..maxHp.." | 📏 "..dist.." | 📡 "..ping.."ms | 🆔 "..targetPlayer.UserId,
            Duration = 6,
        })
    end,
})

PlayersTab:CreateButton({
    Name = "Teleportarme al jugador",
    Callback = function()
        if not targetPlayer then
            Rayfield:Notify({ Title = "Error", Content = "Selecciona un jugador primero", Duration = 3 })
            return
        end
        local hrp = lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
        local targetHrp = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if hrp and targetHrp then
            hrp.CFrame = targetHrp.CFrame + Vector3.new(3, 0, 0)
            Rayfield:Notify({ Title = "✅ Teleportado", Content = "Fuiste a "..targetPlayer.Name, Duration = 2 })
        end
    end,
})

PlayersTab:CreateToggle({
    Name = "Seguir jugador",
    CurrentValue = false,
    Flag = "Follow",
    Callback = function(val)
        limpiarConexion("seguir")
        if val and targetPlayer then
            conexiones["seguir"] = RunService.Heartbeat:Connect(function()
                local hrp = lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
                local targetHrp = targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                if hrp and targetHrp then
                    if (hrp.Position - targetHrp.Position).Magnitude > 5 then
                        hrp.CFrame = targetHrp.CFrame + Vector3.new(3, 0, 0)

                end

    end,
})

PlayersTab:CreateToggle({
    Name = "Cámara apuntando al jugador",
    CurrentValue = false,
    Flag = "CamLock",
    Callback = function(val)
        limpiarConexion("camara")
        if val and targetPlayer then
            conexiones["camara"] = RunService.RenderStepped:Connect(function()
                local targetChar = targetPlayer.Character
                if targetChar and targetChar:FindFirstChild("HumanoidRootPart") then
                    cam.CFrame = CFrame.lookAt(cam.CFrame.Position, targetChar.HumanoidRootPart.Position)
                end
            end)
        end
    end,
})

PlayersTab:CreateToggle({
    Name = "ESP todos los jugadores",
    CurrentValue = false,
    Flag = "ESP",
    Callback = function(val)
        local function addESP(player)
            if player == lp then return end
            local function applyHL(char)
                task.wait(0.5)
                local old = char:FindFirstChild("ESP_HL")
                if old then old:Destroy() end
                if val then
                    local h = Instance.new("Highlight")
                    h.Name = "ESP_HL"
                    h.FillColor = Color3.fromRGB(255, 50, 50)
                    h.OutlineColor = Color3.fromRGB(255, 255, 255)
                    h.Parent = char
                end
            end
            player.CharacterAdded:Connect(applyHL)
            if player.Character then applyHL(player.Character) end
        end
        for _, p in ipairs(Players:GetPlayers()) do addESP(p) end
        Players.PlayerAdded:Connect(addESP)
    end,
})

PlayersTab:CreateToggle({
    Name = "X-Ray (ver a través de paredes)",
    CurrentValue = false,
    Flag = "XRay",
    Callback = function(val)
        limpiarConexion("xray")

        local function aplicarXray(player)
            if player == lp then return end
            local function ponerHL(char)
                task.wait(0.3)
                if not char then return end
                local old = char:FindFirstChild("XRAY_HL")
                if old then old:Destroy() end
                if val then
                    local hl = Instance.new("Highlight")
                    hl.Name = "XRAY_HL"
                    hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                    hl.FillColor = Color3.fromRGB(0, 200, 255)
                    hl.OutlineColor = Color3.fromRGB(255, 255, 255)
                    hl.FillTransparency = 0.4
                    hl.Parent = char
                end
            end
            if player.Character then ponerHL(player.Character) end
            player.CharacterAdded:Connect(function(char)
                if val then ponerHL(char) end
            end)
        end

        if val then
            for _, p in ipairs(Players:GetPlayers()) do aplicarXray(p) end
            conexiones["xray"] = Players.PlayerAdded:Connect(function(p) aplicarXray(p) end)
        else
            for _, p in ipairs(Players:GetPlayers()) do
                if p.Character then
                    local h = p.Character:FindFirstChild("XRAY_HL")
                    if h then h:Destroy() end
                end
            end
        end
    end,
})
